import '../styles/KanbanCardContainer.css';
import React, { useState, useEffect, useRef } from 'react';
import KanbanCard from './KanbanCard';
import { Droppable } from './Droppable';
import {DndContext} from '@dnd-kit/core';
import Draggable from './Draggable';


const KanbanCardContainer = () => {
   
    const [parent, setParent] = useState(null);

    useEffect(()=>{
        setParent("droppable1");
    },[])

    function handleDragEnd({over}) {
        // If the item is dropped over a container, set it as the parent
        // otherwise reset the parent to `null`
        console.log(over.id);
        setParent(over ? over.id : null);

    }

    const draggable = (
         <Draggable key="card1" id="card1" >
                <KanbanCard />
        </Draggable>
    );

    return (
        <div style={{display: 'flex'}}>
            <DndContext onDragEnd={handleDragEnd}>
                <Droppable key="droppable1" id="droppable1" >
                    <div className='KanbanCardContainer'>
                        {parent === "droppable1" ? 
                            draggable : "drop here"
                        } 
                    </div>
                </Droppable>
                <Droppable key="droppable2" id="droppable2">
                    <div className='KanbanCardContainer'>
                        {parent === "droppable2" ? 
                            draggable : "drop here"
                        } 
                    </div>
                </Droppable>
            </DndContext>
        </div>
    );
}
export default KanbanCardContainer;
